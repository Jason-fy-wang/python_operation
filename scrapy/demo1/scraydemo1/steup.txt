# 创建 一个工厂
scrapy startproject scraydemo1

# 创建一个 spider
scrapy genspider example example.com

# 列出当前可用的爬虫
scrapy list

# 运行一个爬虫
scrapy crawl spdemo1

# 运行爬虫,并输出结果到一个文件中
scrapy crawl spdemo1 -o data.json # 直接输出json格式的文件
scrapy crawl spdemo1 -o data.jsonl # 直接输出json格式的文件
scrapy crawl spdemo1 -o data.csv # 直接输出csv格式的文件
scrapy crawl spdemo1 -o data.xml  #直接输出xml格式文件
